CREATE TABLE Record
(
	ID INTEGER PRIMARY KEY AUTOINCREMENT,
	MemberID INT NOT NULL,
	Name TEXT NOT NULL,
	Medal TINYINT NOT NULL,
	Comment TEXT NOT NULL,
	Exp INT NOT NULL,
	RecTime BIGINT NOT NULL, 
	Senka SMALLINT NULL DEFAULT 0,
	EOSenka SMALLINT NULL DEFAULT 0
);

CREATE TABLE Inherit
(
	ID INTEGER PRIMARY KEY AUTOINCREMENT,
	MemberID INT NOT NULL,
	InheritSenka SMALLINT,
	Month TINYINT NOT NULL
);
