IF EXISTS (SELECT 1 FROM SYSOBJECTS WHERE name = 'Member')
DROP TABLE Member

CREATE TABLE Member
(
	ID INT NOT NULL IDENTITY(1,1),
	MemberID NVARCHAR(10)NOT NULL,
	Name NVARCHAR(50) COLLATE Japanese_CS_AS NOT NULL,
	CurrentExp INT NOT NULL,
	YearBeginExp INT NOT NULL,
	MonthBeginExp INT NOT NULL,
	CurrentComment NVARCHAR(50) COLLATE Japanese_CS_AS NOT NULL,
	CurrentMedal INT NULL,
	LastMonthBonus INT NOT NULL,
	CurrSeasonBonusTotal INT NOT NULL,
	UpdateTimeStamp NVARCHAR(50) NOT NULL,
	PRIMARY KEY(ID)
)
